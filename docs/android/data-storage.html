                import androidx.appcompat.app.AppCompatActivity;

import android.content.Context;
import android.content.SharedPreferences;
import android.os.Bundle;
import android.view.View;
import android.widget.CheckBox;
import android.widget.EditText;
import android.widget.LinearLayout;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.Toast;

public class SharedPreActivity extends AppCompatActivity implements RadioGroup.OnCheckedChangeListener, View.OnClickListener {

    private EditText etName;
    private EditText etAge;
    private RadioGroup rgSex;
    private LinearLayout llHobby;
    private String gender;
    private SharedPreferences userInfo;
    private SharedPreferences.Editor editor;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_shared_preferences);
        etName = findViewById(R.id.name);
        etAge = findViewById(R.id.age);

        rgSex = findViewById(R.id.rg_sex);
        llHobby = findViewById(R.id.ll_hobby);

        rgSex.setOnCheckedChangeListener(this);
        findViewById(R.id.submit).setOnClickListener(this);


        userInfo = getSharedPreferences("user_info", Context.MODE_PRIVATE);

        loadUserInfo();
    }

    @Override
    public void onCheckedChanged(RadioGroup group, int checkedId) {
        gender = "";

        if (checkedId == R.id.rb_male) {
            gender = "男";
        }
        if (checkedId == R.id.rb_female) {
            gender = "女";
        }
        if (checkedId == R.id.rb_secret) {
            gender = "保密";
        }
    }


    @Override
    public void onClick(View v) {
        if (v.getId() == R.id.submit) {
            submit();
        }
    }

    private void loadUserInfo() {
        String name = userInfo.getString("name", "");
        int age = userInfo.getInt("age", 0);
        String sex = userInfo.getString("gender", "");
        String hobby = userInfo.getString("hobby", "");

        if (!name.equals("")) {
            etName.setText(name);
            etAge.setText(age != 0 ? String.valueOf(age) : "");

            // 设置性别
            for (int i = 0; i < rgSex.getChildCount(); i++) {
                View child = rgSex.getChildAt(i);
                if (child instanceof RadioButton) {
                    RadioButton rb = (RadioButton) child;
                    if (rb.getText().equals(sex)) {
                        rb.setChecked(true);
                    }
                }
            }

            // 设置兴趣爱好
            for (int i = 0; i < llHobby.getChildCount(); i++) {
                View child = llHobby.getChildAt(i);
                if (child instanceof CheckBox) {
                    CheckBox cb = (CheckBox) child;
                    if (hobby.contains(cb.getText().toString())) {
                        cb.setChecked(true);
                    }
                }
            }
        }
    }

    private void submit() {
        // 获取已经选择的兴趣爱好
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < llHobby.getChildCount(); i++) {
            View child = llHobby.getChildAt(i);
            if (child instanceof CheckBox) {
                CheckBox cb = (CheckBox) child;
                if (cb.isChecked()) {
                    sb.append(cb.getText().toString()).append(",");
                }
            }
        }

        String etName = this.etName.getText().toString();
        String etAge = this.etAge.getText().toString();

        editor = userInfo.edit();

        editor.putString("name", etName);
        editor.putInt("age", Integer.parseInt(etAge));
        editor.putString("gender", gender);
        editor.putString("hobby", sb.toString());
        // 提交
        editor.apply();

        Toast.makeText(this, "保存成功", Toast.LENGTH_SHORT).show();
    }
}

                # SQLiteDatabase

                SQLiteOpenHelper是Android提供的数据库辅助工具，用于指导开发者进行SQLite的合理使用。

                SQLiteOpenHelper的具体使用步骤如下:

                • 新建一个继承自SQLiteOpenHelper的数据库操作类，提示重写onCreate和onUpgrade两个方 法。

                • 封装保证数据库安全的必要方法。

                • 提供对表记录进行增加、删除、修改、查询的操作方法。

                public class SQLiteHelperActivity extends AppCompatActivity implements View.OnClickListener {
    private EditText etName;
    private EditText etAge;
    private RadioGroup rgSex;
    private LinearLayout llHobby;
    private UserDBHelper userDBHelper;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_sqlite_helper);

        etName = findViewById(R.id.name);
        etAge = findViewById(R.id.age);

        rgSex = findViewById(R.id.rg_sex);
        llHobby = findViewById(R.id.ll_hobby);

        findViewById(R.id.btn_submit).setOnClickListener(this);
        findViewById(R.id.btn_delete).setOnClickListener(this);
        findViewById(R.id.btn_update).setOnClickListener(this);
        findViewById(R.id.btn_query).setOnClickListener(this);
    }


    @Override
    protected void onStart() {
        super.onStart();
        userDBHelper = UserDBHelper.getInstance(this);
        userDBHelper.openReadableDatabase();
        userDBHelper.openWritableDatabase();
    }

    @Override
    protected void onStop() {
        super.onStop();
        userDBHelper.closeDatabase();
    }

    @Override
    public void onClick(View v) {
        String username = this.etName.getText() == null ? "" : this.etName.getText().toString();
        int userAge = Integer.parseInt((this.etAge.getText().toString().equals("")) ? "0" : this.etAge.getText().toString());
        String gender = "";
        StringBuilder hobby = new StringBuilder();

        for (int i = 0; i < llHobby.getChildCount(); i++) {
            View child = llHobby.getChildAt(i);
            if (child instanceof CheckBox) {
                CheckBox cb = (CheckBox) child;
                if (cb.isChecked()) {
                    hobby.append(cb.getText().toString()).append(" ");
                }
            }
        }
        for (int i = 0; i < rgSex.getChildCount(); i++) {
            View child = rgSex.getChildAt(i);
            if (child instanceof RadioButton) {
                RadioButton cb = (RadioButton) child;
                if (cb.isChecked()) {
                    gender = cb.getText().toString();
                }
            }
        }
        User user = new User(username,
                userAge,
                gender,
                hobby.toString());

        if (v.getId() == R.id.btn_submit) {
            addUser(user);
        }
        if (v.getId() == R.id.btn_delete) {
            deleteUser(user);
        }
        if (v.getId() == R.id.btn_update) {
            updateUser(user);
        }
        if (v.getId() == R.id.btn_query) {
            queryUser();
        }
    }

    private void queryUser() {
        List<User> userList = userDBHelper.queryUserList();
        if (userList != null && userList.size() > 0) {
            for (User u : userList) {
              	// 查询
                Log.d("SQLite", u.toString());
            }
        }
    }


    private void deleteUser(User user) {
        int i = userDBHelper.deleteUser(user);
        if (i > 0)
            Toast.makeText(this, "删除成功", Toast.LENGTH_SHORT).show();
    }

    private void addUser(User user) {
        long insertUser = userDBHelper.insertUser(user);
        if (insertUser > 0)

        int i = userDBHelper.updateUser(user);

                // 数据库交互
public class UserDBHelper extends SQLiteOpenHelper {

    private static UserDBHelper dbHelper;

    public static UserDBHelper getInstance(Context context) {
        if (dbHelper == null) {
            synchronized (UserDBHelper.class) {
                if (dbHelper == null) {
                    dbHelper = new UserDBHelper(context);
                }
            }
        }
        return dbHelper;
    }

    private SQLiteDatabase ReadDB;
    private SQLiteDatabase WriteDB;

    public SQLiteDatabase openWritableDatabase() {
        if (WriteDB == null || !WriteDB.isOpen()) {
            WriteDB = getWritableDatabase();
        }
        return WriteDB;
    }

    public SQLiteDatabase openReadableDatabase() {
        if (ReadDB == null || !ReadDB.isOpen()) {
            ReadDB = getReadableDatabase();
        }
        return ReadDB;
    }

    public void closeDatabase() {
        if (ReadDB != null && ReadDB.isOpen()) {
            ReadDB.close();
            ReadDB = null;
        }
        if (WriteDB != null && WriteDB.isOpen()) {
            WriteDB.close();
            ReadDB = null;
        }
    }


    private static final String DATABASE_NAME = "user.db";

    private static final int DATABASE_VERSION = 1;

    public UserDBHelper(@Nullable Context context) {
        super(context, DATABASE_NAME, null, DATABASE_VERSION);
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        db.execSQL(UserContract.SQL_CREATE_USER_ENTRIES);
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {

    }

    public long insertUser(User user) {
        // nullColumnHack: null 表示如果 ContentValues 为空，将不会插入
        return WriteDB.insert(UserContract.UserTable.TABLE_NAME, null, user.toContentValues());
    }

    public int deleteUser(User user) {
        return WriteDB.delete(UserContract.UserTable.TABLE_NAME, "username=?", new String[]{user.getUsername()});
    }

    public int updateUser(User user) {
        return WriteDB.update(UserContract.UserTable.TABLE_NAME, user.toContentValues(), "username=?", new String[]{user.getUsername()});
    }

    public List<User> queryUserList() {
        ArrayList<User> userList = new ArrayList<>();
        Cursor cursor = ReadDB.query(UserContract.UserTable.TABLE_NAME, null, null, null, null, null, null);
        while (cursor.moveToNext()) {
            User user = new User();
            user.setUserId(cursor.getInt(0));
            user.setUsername(cursor.getString(1));
            user.setAge(cursor.getInt(2));
            user.setGender(cursor.getString(3));
            user.setHobby(cursor.getString(4));
            userList.add(user);
        }
        cursor.close();
        return userList;
    }
}

                // 数据库表字段
public class UserContract {

    private UserContract() {
    }

		public static final String SQL_CREATE_USER_ENTRIES =
            "CREATE TABLE IF NOT EXISTS" + UserTable.TABLE_NAME + " (" +
                    UserTable._ID + " INTEGER PRIMARY KEY," +
                    UserTable.COLUMN_NAME_USERNAME + " TEXT," +
                    UserTable.COLUMN_NAME_AGE + " INTEGER," +
                    UserTable.COLUMN_NAME_GENDER + " TEXT," +
                    UserTable.COLUMN_NAME_HOBBY + " TEXT)";

    public static final String SQL_DELETE_USER_ENTRIES =
            "DROP TABLE IF EXISTS " + UserTable.TABLE_NAME;

    public static class UserTable {

        public static final String TABLE_NAME = "user_table";
        public static final String _ID = "user_id";
        public static final String COLUMN_NAME_USERNAME = "username";
        public static final String COLUMN_NAME_AGE = "age";
        public static final String COLUMN_NAME_GENDER = "gender";
        public static final String COLUMN_NAME_HOBBY = "hobby";
    }
}

                // 实体类
public class User {

    private int userId;
    private String username;
    private int age;
    private String gender;
    private String hobby;

    public User() {
    }

    public User(String username, int age, String gender, String hobby) {
        this.username = username;
        this.age = age;
        this.gender = gender;
        this.hobby = hobby;
    }

    public int getUserId() {
        return userId;
    }

    public void setUserId(int userId) {
        this.userId = userId;
    }

    public String getUsername() {
        return username;
    }

    public void setUsername(String username) {
        this.username = username;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public String getGender() {
        return gender;
    }

    public void setGender(String gender) {
        this.gender = gender;
    }

    public String getHobby() {
        return hobby;
    }

    public void setHobby(String hobby) {
        this.hobby = hobby;
    }

    @Override
    public String toString() {
        return "User{" +
                "userId=" + userId +
                ", username='" + username + '\'' +
                ", age=" + age +
                ", gender='" + gender + '\'' +
                ", hobby='" + hobby + '\'' +
                '}';
    }

    public ContentValues toContentValues() {
        ContentValues values = new ContentValues();
        values.put(UserContract.UserTable.COLUMN_NAME_USERNAME, username);
        values.put(UserContract.UserTable.COLUMN_NAME_AGE, age);
        values.put(UserContract.UserTable.COLUMN_NAME_GENDER, gender);
        values.put(UserContract.UserTable.COLUMN_NAME_HOBBY, hobby);
        return values;
    }
}
