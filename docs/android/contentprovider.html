                    # Uri

                    ContentProvider 使用的Uri 语法结构如下:

                    • content://authority/data_path/id
                    • 「content://」 是通用前缀，表示该Uri用于ContentProvider定位资源。
                    • 「authority」 是授权者名称，用来确定具体由哪一个ContentProvider提供资源。因此一般authority都由

                    类的小写全称组成，以保证唯一性。

                    • 「data_path」 是数据路径，用来确定请求的是哪个数据集(xxx.db)。
                    • 「id」 是数据编号，用来请求单条数据。如果是多条这个字段忽略。

                    # authority

                    作为Provider应用，需要在AndroidManifest.xml中添加代码，并且建议将authorities设置为com.example.xxxxxx.provider类名

                    <manifest...>
    <application...>
      	// Code for activity...
      
      	
        <provider
            android:name=".provider.UserInfoContentProvider"
            android:authorities="包名全路径.XXXContentProvider"
            android:enabled="true"
            android:exported="true"/>
    </application>
</manifest>

                    新建一个RoomDatabase类

                    @Database(entities = {User.class}, version = 1)
public abstract class AppRoomDataBase extends RoomDatabase {

  	// 获取 用户表操作
    public abstract UserDao getUserDao();

    private static AppRoomDataBase INSTANCE;

    public static AppRoomDataBase getDatabase(Context applicationContext) {
        if (INSTANCE == null) {
            synchronized (AppRoomDataBase.class) {
                if (INSTANCE == null) {
                  	// 创建数据库
                    INSTANCE = Room.databaseBuilder(applicationContext, AppRoomDataBase.class, "user_database.db")
                            .allowMainThreadQueries()
                            .build();
                }
            }
        }
        return INSTANCE;
    }

}

        this.id = id;
    }

    public String getUsername() {
        return username;

    }

        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public double getHeight() {
        return height;
    }

    public void setHeight(double height) {
        this.height = height;
    }

    public int getGender() {
        return gender;
    }

    public void setGender(int gender) {
        this.gender = gender;
    }

    public Long getBirthday() {
        return birthday;
    }

    public void setBirthday(Long birthday) {
        this.birthday = birthday;
    }

    @Override
    public String toString() {
        return "User{" +
                "id=" + id +
                ", username='" + username + '\'' +
                ", age=" + age +
                ", height=" + height +
                ", gender=" + gender +

                '}';
    }

    public void setValues(ContentValues values) {
        this.username = values.getAsString("username");
        this.age = values.getAsInteger("age");
        this.height = values.getAsDouble("height");
        this.gender = values.getAsInteger("gender");
        this.birthday = values.getAsLong("birthday");
    }
}

public class UserInfoContentProvider extends ContentProvider {

    private AppRoomDataBase db;
    private UserDao userDao;

    private static final UriMatcher uriMatcher = new UriMatcher(UriMatcher.NO_MATCH);

    public UserInfoContentProvider() {
      	// 设置查询路径，并定义 Code
        uriMatcher.addURI("com.wesley.application06_provider.provider.UserInfoContentProvider", "tb_user/#", 1);
    }

    @Override
    public boolean onCreate() {
        db = AppRoomDataBase.getDatabase(getContext().getApplicationContext());
        userDao = db.getUserDao();
        return true;
    }

    @Override
    public int delete(Uri uri, String selection, String[] selectionArgs) {
        // Implement this to handle requests to delete one or more rows.
        throw new UnsupportedOperationException("Not yet implemented");
    }

    @Override
    public String getType(Uri uri) {
        // TODO: Implement this to handle requests for the MIME type of the data
        // at the given URI.
        throw new UnsupportedOperationException("Not yet implemented");
    }

    @Override
    public Uri insert(Uri uri, ContentValues values) {
        User user = new User();
        user.setValues(values);
        userDao.insert(user);
        return uri;
    }

    @Override
    public Cursor query(Uri uri, String[] projection, String selection,
                        String[] selectionArgs, String sortOrder) {
        int match = uriMatcher.match(uri);
        switch (match) {
            // 寻找 code 为 1 的 Uri
            case 1:
                String id = uri.getLastPathSegment();
                User user = userDao.queryById(Integer.parseInt(id));
                if (user != null) {
                    MatrixCursor cursor = new MatrixCursor(new String[]{"_id", "username", "age", "height", "gender", "birthday"});
                    cursor.addRow(new Object[]{user.getId(), user.getUsername(), user.getAge(), user.getHeight(), user.getGender(), user.getBirthday()});
                    return cursor;
                }
                break;
            default:
                throw new IllegalArgumentException("Unknown URI " + uri);
        }
        return null;
    }

    @Override
    public int update(Uri uri, ContentValues values, String selection,
                      String[] selectionArgs) {
        // TODO: Implement this to handle requests to update one or more rows.
        throw new UnsupportedOperationException("Not yet implemented");
    }
}

                    # Client 客户端

                    public class MainActivity extends AppCompatActivity implements View.OnClickListener {

    private Button btnBirthday;
    private View datePickerView;

    private AlertDialog datePickerDialog;
    private EditText etName;
    private EditText etAge;
    private EditText etHeight;
    private RadioButton rbMale;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.et_name);
        etAge = findViewById(R.id.et_age);
        etHeight = findViewById(R.id.et_height);
        rbMale = findViewById(R.id.rb_male);

        LayoutInflater inflater = LayoutInflater.from(this);
        datePickerView = inflater.inflate(R.layout.date_picker, null);
        datePickerView.findViewById(R.id.btn_date_picker).setOnClickListener(this);

        btnBirthday = findViewById(R.id.btn_birthday);
        btnBirthday.setOnClickListener(this);

        findViewById(R.id.btn_submit).setOnClickListener(this);
        findViewById(R.id.btn_query).setOnClickListener(this);
    }

    @Override
    protected void onStart() {
        super.onStart();
        Calendar calendar = Calendar.getInstance();
        int year = calendar.get(Calendar.YEAR);
        int month = calendar.get(Calendar.MONTH);
        int dayOfMonth = calendar.get(Calendar.DAY_OF_MONTH);
        btnBirthday.setText(year + "-" + (month + 1) + "-" + dayOfMonth);
    }

    @SuppressLint("Range")
    @Override
    public void onClick(View v) {
        if (v.getId() == R.id.btn_birthday) {
            if (datePickerDialog != null && !datePickerDialog.isShowing()) {
                datePickerDialog.show();
            } else {
                AlertDialog.Builder builder = new AlertDialog.Builder(this);
                builder.setTitle("选择生日");
                builder.setView(datePickerView);
                datePickerDialog = builder.create();
                datePickerDialog.show();
            }
        }
        if (v.getId() == R.id.btn_date_picker) {
            DatePicker datePicker = datePickerView.findViewById(R.id.date_picker);
            int year = datePicker.getYear();
            int month = datePicker.getMonth();
            int day = datePicker.getDayOfMonth();
            btnBirthday.setText(year + "-" + (month + 1) + "-" + day);
            datePickerDialog.dismiss();
        }

        if (v.getId() == R.id.btn_submit) {

            Calendar calendar = Calendar.getInstance();
            String birthday = btnBirthday.getText().toString();
            ContentValues values = new ContentValues();
            calendar.set(Integer.parseInt(birthday.split("-")[0]), Integer.parseInt(birthday.split("-")[1]) - 1, Integer.parseInt(birthday.split("-")[2]));

            values.put(UserInfoContent.USER_NAME, etName.getText().toString());
            values.put(UserInfoContent.USER_AGE, etAge.getText().toString());
            values.put(UserInfoContent.USER_HEIGHT, etHeight.getText().toString());
            values.put(UserInfoContent.USER_GENDER, rbMale.isChecked() ? 1 : 0);
            values.put(UserInfoContent.USER_BIRTHDAY, calendar.getTimeInMillis());

            getContentResolver().insert(UserInfoContent.CONTENT_URI, values);
            Toast.makeText(this, "提交成功", Toast.LENGTH_SHORT).show();
        }

        if (v.getId() == R.id.btn_query) {
            Uri uri = ContentUris.withAppendedId(UserInfoContent.CONTENT_URI, 1);
            Cursor cursor = getContentResolver().query(uri, null, null, null, null);
            if (cursor != null) {
                User user = new User();
                while (cursor.moveToNext()) {
                    user.setId(cursor.getInt(cursor.getColumnIndex(UserInfoContent._ID)));
                    user.setUsername(cursor.getString(cursor.getColumnIndex(UserInfoContent.USER_NAME)));
                    user.setAge(cursor.getInt(cursor.getColumnIndex(UserInfoContent.USER_AGE)));
                    user.setHeight(cursor.getDouble(cursor.getColumnIndex(UserInfoContent.USER_HEIGHT)));
                    user.setGender(cursor.getInt(cursor.getColumnIndex(UserInfoContent.USER_GENDER)));
                    user.setBirthday(cursor.getLong(cursor.getColumnIndex(UserInfoContent.USER_BIRTHDAY)));
                }
                Log.d("MainActivity", user.toString());
                cursor.close();
            }

        }
    }
}
