                          # Drawble

                          在 Android 应用开发中，drawable 是用于存储图形资源的文件夹之一。它可以包含多种类型的图像资源，如位图、矢量图形和图层列表。

                          # 9-Patch

                          在Android Studio中右键原图创建9-Patch图片文件

                          左、上侧为可拉伸的区域

                          右、下侧是文字覆盖区域

                          # selector

                          selector 可以帮助开发人员在应用程序中创建具有不同状态的用户界面元素，例如按钮、文本视图、图像视图等。

                          简单示例：

                          <selector xmlns:android="http://schemas.android.com/apk/res/android">
  	<!--如果按钮被按下，使用 @drawable/button_pressed 图片-->
    <item android:drawable="@drawable/button_pressed" android:state_pressed="true"/>
 	  <!--如果按钮获得焦点，使用@drawable/button_focused  图片-->
    <item android:drawable="@drawable/button_focused" android:state_focused="true"/>
 	  <!--否则使用 @drawable/button_normal 图片-->
    <item android:drawable="@drawable/button_normal"/>
</selector>

                          # shape

                          shape 可以帮助开发人员在应用程序中创建不同形状的用户界面元素，例如按钮、文本视图、图像视图等。

                          简单示例：

                          <?xml version="1.0" encoding="utf-8"?>
<shape xmlns:android="http://schemas.android.com/apk/res/android"
    android:shape="oval">

    <!--填充-->
    <solid android:color="#ff66aa"/>

    <!--边框-->
    <stroke android:width="1dp"
        android:color="#aaaaaa"/>

</shape>

                          <?xml version="1.0" encoding="utf-8"?>
<shape xmlns:android="http://schemas.android.com/apk/res/android">

    <solid android:color="#ffdd66"/>

    <stroke android:width="1dp"
        android:color="#aaaaaa"/>

    <corners android:radius="10dp"/>

</shape>

                          # CheckBox

                          <?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    android:orientation="vertical">

    <CheckBox
        android:id="@+id/check_box_default"
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="This is default CheckBox" />

    <CheckBox
        android:id="@+id/check_box_custom"
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:button="@drawable/check_box_selector"
        android:text="This is CheckBox with custom style" />

    <!--使用CheckBox代替Switch，可以获得更好的兼容-->
    <LinearLayout
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:orientation="horizontal">

        <TextView
            android:id="@+id/switch_check_box_text"
            android:layout_width="0dp"
            android:layout_height="wrap_content"
            android:layout_gravity="start"
            android:layout_weight="1"
            android:padding="15dp"
            android:text="Switch custom" />

        <CheckBox
            android:id="@+id/check_box_switch"
            android:layout_width="wrap_content"
            android:layout_height="wrap_content"
            android:background="@drawable/switch_selector"
            android:button="@null"
            android:gravity="end" />
    </LinearLayout>
</LinearLayout>

                          public class CheckBoxActivity extends AppCompatActivity implements CompoundButton.OnCheckedChangeListener {

    private CheckBox boxDefault;
    private CheckBox boxCustom;
    private CheckBox boxSwitch;
    private TextView switchText;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_check_box);

        switchText = findViewById(R.id.switch_check_box_text);

        boxDefault = findViewById(R.id.check_box_default);
        boxCustom = findViewById(R.id.check_box_custom);
        boxSwitch = findViewById(R.id.check_box_switch);

        boxDefault.setOnCheckedChangeListener(this);
        boxCustom.setOnCheckedChangeListener(this);
        boxSwitch.setOnCheckedChangeListener(this);

        switchText.setText("Switch custom: " + (boxSwitch.isChecked() ? "ON" : "OFF"));
    }

    @Override
    public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
        String msg = "Status: " + (isChecked ? "Checked" : "Unchecked");
        if (buttonView.getId() == R.id.check_box_default) {
            boxDefault.setText(msg);
        } else if (buttonView.getId() == R.id.check_box_custom) {
            boxCustom.setText(msg);
        }

        if (buttonView.getId() == R.id.check_box_switch) {
            switchText.setText("Switch custom: " + (isChecked ? "ON" : "OFF"));
        }
    }
}

                          public class RadioActivity extends AppCompatActivity implements RadioGroup.OnCheckedChangeListener {

    private RadioGroup rgGender;
    private TextView tvSex;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_radio);

        rgGender = findViewById(R.id.rg_gender);
        tvSex = findViewById(R.id.tv_sex);

        rgGender.setOnCheckedChangeListener(this);
    }

    @Override
    public void onCheckedChanged(RadioGroup group, int checkedId) {
        if (checkedId == R.id.male){
            tvSex.setText("You are handsome!");
        }
        if (checkedId == R.id.female){
            tvSex.setText("You are beautiful!");
        }
        if (checkedId == R.id.secret){
            tvSex.setText("You are special!");
        }
    }
}

                          public class SwitchActivity extends AppCompatActivity implements CompoundButton.OnCheckedChangeListener {

    private Switch switchButton;
    private TextView switchStatus;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_switch);
        switchStatus = findViewById(R.id.switch_status);

        switchButton = findViewById(R.id.switch_default);
        switchButton.setOnCheckedChangeListener(this);
    }

    @Override
    public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
        String status = String.format("Switch is %s", isChecked ? "ON" : "OFF");
        if (buttonView.getId() == R.id.switch_default) {
            if (isChecked) {
                switchStatus.setText(status);
            } else {
                switchStatus.setText(status);
            }
        }
    }
}

                          public class EditTextActivity extends AppCompatActivity implements View.OnFocusChangeListener, TextWatcher {

    private EditText etAccount;
    private EditText etPassword;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_edit_text);

        etAccount = findViewById(R.id.et_account);
        etPassword = findViewById(R.id.et_password);

        // 设置焦点变化的监听
        etAccount.setOnFocusChangeListener(this);
        // 设置内容变化的监听
        etPassword.addTextChangedListener(this);
    }


    @Override
    public void onFocusChange(View v, boolean hasFocus) {
        if (v.getId() == R.id.et_account) {
            if (!hasFocus) {
                if (etAccount.getText().toString().length() > 6) {
                    Toast.makeText(this, "帐号长度大于6位", Toast.LENGTH_SHORT).show();
                    etAccount.setError("帐号长度大于6位");
                }
            }
        }
    }

    @Override
    public void beforeTextChanged(CharSequence s, int start, int count, int after) {
        
    }

    @Override
    public void onTextChanged(CharSequence s, int start, int before, int count) {
        
    }

    @Override
    public void afterTextChanged(Editable s) {
        String password = s.toString();
        if (password.length() >= 6) {
            Object systemService = this.getSystemService(INPUT_METHOD_SERVICE);
            if (systemService != null) {
                // 隐藏输入键盘
                ((InputMethodManager) systemService).hideSoftInputFromWindow(etPassword.getWindowToken(), 0);
            }
        }
    }
}

        android:layout_width="match_parent"

        android:layout_width="match_parent"

        android:text="Show system date picker" />


    <Button
        android:id="@+id/btn_show_time_picker"
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="Show time picker" />

    <Button
        android:id="@+id/btn_show_system_time_picker"
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="Show system time picker" />
</LinearLayout>

                          public class DialogActivity extends AppCompatActivity implements View.OnClickListener {

    private TextView tvDialog;
    private DatePicker datePicker;
    private View datePickerLayout;
    private AlertDialog datePickerDialog;
    private View timePickerLayout;
    private TimePicker timePicker;
    private AlertDialog timePickerDialog;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_dialog);
        LayoutInflater inflater = LayoutInflater.from(this);

        // 获取日期选择器的布局
        datePickerLayout = inflater.inflate(R.layout.date_picker, null);
        // 获取时间选择器的布局
        timePickerLayout = inflater.inflate(R.layout.time_picker, null);

        tvDialog = findViewById(R.id.tv_dialog);
        // 获取日期选择器
        datePicker = datePickerLayout.findViewById(R.id.date_picker);
        // 获取时间选择器
        timePicker = timePickerLayout.findViewById(R.id.time_picker);
        timePicker.setIs24HourView(true);

        findViewById(R.id.btn_dialog).setOnClickListener(this);
        findViewById(R.id.btn_show_picker).setOnClickListener(this);
        findViewById(R.id.btn_show_system_picker).setOnClickListener(this);
        findViewById(R.id.btn_show_time_picker).setOnClickListener(this);
        findViewById(R.id.btn_show_system_time_picker).setOnClickListener(this);

        // 设置日期选择器的确定按钮的点击事件
        datePickerLayout.findViewById(R.id.btn_date_picker).setOnClickListener(this);
        // 设置时间选择器的确定按钮的点击事件
        timePickerLayout.findViewById(R.id.btn_time_picker).setOnClickListener(this);
    }

    @Override
    public void onClick(View v) {
        // 弹出系统对话框
        if (v.getId() == R.id.btn_dialog) {
            AlertDialog.Builder builder = new AlertDialog.Builder(this);
            builder.setTitle("Dialog");
            builder.setMessage("This is a dialog");
            builder.setPositiveButton("OK", (dialog, which) -> {
                tvDialog.setText("You clicked OK");
            });
            builder.setNegativeButton("Cancel", (dialog, which) -> {
                tvDialog.setText("You clicked Cancel");
            });
            builder.show();
        }

        // 弹出自定义日期选择器
        if (v.getId() == R.id.btn_show_picker) {
            if (datePickerDialog != null && !datePickerDialog.isShowing()) {
                datePickerDialog.show();
            } else {
                AlertDialog.Builder builder = new AlertDialog.Builder(this);
                builder.setTitle("DatePicker");
                builder.setView(datePickerLayout);
                datePickerDialog = builder.create();
                datePickerDialog.show();
            }
        }
        if (v.getId() == R.id.btn_date_picker) {
            int year = datePicker.getYear();
            int month = datePicker.getMonth();
            int day = datePicker.getDayOfMonth();
            tvDialog.setText(year + "-" + month + 1 + "-" + day);
            datePickerDialog.dismiss();
        }

        // 弹出自定义时间选择器
        if (v.getId() == R.id.btn_show_time_picker) {
            if (timePickerDialog != null && !timePickerDialog.isShowing()) {
                timePickerDialog.show();
            } else {
                AlertDialog.Builder builder = new AlertDialog.Builder(this);
                builder.setTitle("TimePicker");
                builder.setView(timePickerLayout);
                timePickerDialog = builder.create();
                timePickerDialog.show();
            }
        }
        if (v.getId() == R.id.btn_time_picker) {
            int hour = timePicker.getHour();
            int minute = timePicker.getMinute();
            tvDialog.setText(hour + ":" + minute);
            timePickerDialog.dismiss();
        }


        // 弹出系统日期选择器
        if (v.getId() == R.id.btn_show_system_picker) {
            // 弹出系统日期选择器
            DatePickerDialog datePickerDialog = new DatePickerDialog(this);
            datePickerDialog.setOnDateSetListener((view, year, month, dayOfMonth) -> {
                tvDialog.setText(year + "-" + month + 1 + "-" + dayOfMonth);
            });
            datePickerDialog.show();
        }

        // 弹出系统时间选择器
        if (v.getId() == R.id.btn_show_system_time_picker) {
            // 弹出系统时间选择器
            TimePickerDialog timePickerDialog = new TimePickerDialog(this,(view, hourOfDay, minute) -> {
                view.setIs24HourView(true);
                tvDialog.setText(hourOfDay + ":" + minute);
            }, 0, 0, true);
            timePickerDialog.show();
        }
    }
}
