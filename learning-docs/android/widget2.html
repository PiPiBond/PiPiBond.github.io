                          public class CheckBoxActivity extends AppCompatActivity implements CompoundButton.OnCheckedChangeListener {

    private CheckBox boxDefault;
    private CheckBox boxCustom;
    private CheckBox boxSwitch;
    private TextView switchText;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_check_box);

        switchText = findViewById(R.id.switch_check_box_text);

        boxDefault = findViewById(R.id.check_box_default);
        boxCustom = findViewById(R.id.check_box_custom);
        boxSwitch = findViewById(R.id.check_box_switch);

        boxDefault.setOnCheckedChangeListener(this);
        boxCustom.setOnCheckedChangeListener(this);
        boxSwitch.setOnCheckedChangeListener(this);

        switchText.setText("Switch custom: " + (boxSwitch.isChecked() ? "ON" : "OFF"));
    }

    @Override
    public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
        String msg = "Status: " + (isChecked ? "Checked" : "Unchecked");
        if (buttonView.getId() == R.id.check_box_default) {
            boxDefault.setText(msg);
        } else if (buttonView.getId() == R.id.check_box_custom) {
            boxCustom.setText(msg);
        }

        if (buttonView.getId() == R.id.check_box_switch) {
            switchText.setText("Switch custom: " + (isChecked ? "ON" : "OFF"));
        }
    }
}

    @Override
    protected void onCreate(Bundle savedInstanceState) {

        setContentView(R.layout.activity_dialog);
        LayoutInflater inflater = LayoutInflater.from(this);

        // è·å–æ—¥æœŸé€‰æ‹©å™¨çš„å¸ƒå±€

        datePicker = datePickerLayout.findViewById(R.id.date_picker);
        // è·å–æ—¶é—´é€‰æ‹©å™¨
        timePicker = timePickerLayout.findViewById(R.id.time_picker);
        timePicker.setIs24HourView(true);

        findViewById(R.id.btn_dialog).setOnClickListener(this);
        findViewById(R.id.btn_show_picker).setOnClickListener(this);
        findViewById(R.id.btn_show_system_picker).setOnClickListener(this);
        findViewById(R.id.btn_show_time_picker).setOnClickListener(this);
        findViewById(R.id.btn_show_system_time_picker).setOnClickListener(this);

        // è®¾ç½®æ—¥æœŸé€‰æ‹©å™¨çš„ç¡®å®šæŒ‰é’®çš„ç‚¹å‡»äº‹ä»¶
        datePickerLayout.findViewById(R.id.btn_date_picker).setOnClickListener(this);
        // è®¾ç½®æ—¶é—´é€‰æ‹©å™¨çš„ç¡®å®šæŒ‰é’®çš„ç‚¹å‡»äº‹ä»¶
        timePickerLayout.findViewById(R.id.btn_time_picker).setOnClickListener(this);
    }

    @Override
    public void onClick(View v) {
        // å¼¹å‡ºç³»ç»Ÿå¯¹è¯æ¡†
        if (v.getId() == R.id.btn_dialog) {
            AlertDialog.Builder builder = new AlertDialog.Builder(this);
            builder.setTitle("Dialog");
            builder.setMessage("This is a dialog");
            builder.setPositiveButton("OK", (dialog, which) -> {
                tvDialog.setText("You clicked OK");
            });
            builder.setNegativeButton("Cancel", (dialog, which) -> {
                tvDialog.setText("You clicked Cancel");
            });
            builder.show();
        }

        // å¼¹å‡ºè‡ªå®šä¹‰æ—¥æœŸé€‰æ‹©å™¨
        if (v.getId() == R.id.btn_show_picker) {
            if (datePickerDialog != null && !datePickerDialog.isShowing()) {
                datePickerDialog.show();
            } else {
                AlertDialog.Builder builder = new AlertDialog.Builder(this);
                builder.setTitle("DatePicker");
                builder.setView(datePickerLayout);
                datePickerDialog = builder.create();
                datePickerDialog.show();
            }
        }
        if (v.getId() == R.id.btn_date_picker) {
            int year = datePicker.getYear();
            int month = datePicker.getMonth();
            int day = datePicker.getDayOfMonth();
            tvDialog.setText(year + "-" + month + 1 + "-" + day);
            datePickerDialog.dismiss();
        }

        // å¼¹å‡ºè‡ªå®šä¹‰æ—¶é—´é€‰æ‹©å™¨
