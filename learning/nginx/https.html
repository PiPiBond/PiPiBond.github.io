                  # 描述

                  HTTPS (Hypertext Transfer Protocol Secure)是 HTTP 的安全版本。它使用 SSL/TLS (安全套接字层/传输层安全)对 web 服务器和客户端之间传输的数据进行加密，使攻击者更难以拦截、读取或修改数据。HTTPS 广泛用于互联网上的安全通信，特别是在传输密码、信用卡详细信息或个人信息等敏感信息时。

                  # 生成 JKS 格式密钥库

                  打开 Windows 终端

                  在终端键入

                  keytool -genkeypair -alias <要处理的条目的别名> -keyalg <密钥算法名称>  -keysize <密钥位大小> -keypass <密钥口令> -validity <有效天数> -keystore <密钥库名称> -storepass <密钥库口令>

                  替换 <> 里的内容，并回车

                  keytool -genkeypair -alias vuedemo -keyalg RSA  -keysize 2048 -keypass 123456 -validity 3660 -keystore vuedemo.jks -storepass 123456

                  # 将 JKS 转为 P 12 密钥库

                  在终端键入

                  keytool -importkeystore -srckeystore <源密钥库名称> -destkeystore <目标密钥库名称> -deststoretype <目标密钥库类型>

                  替换 <> 里的内容，并回车

                  keytool -importkeystore -srckeystore vuedemo.jks -destkeystore vuedemo.p12 -deststoretype pkcs12

                  # 生成证书

                  注意替换为自己的名称

                  openssl pkcs12 -in vuedemo.p12 -nokeys -clcerts -out vuedemo.crt
#Enter Import Password:

                  openssl pkcs12 -in vuedemo.p12 -nocerts -nodes -out vuedemo.key
#Enter Import Password:

                  证书生成后移动到 nginx 根目录的 conf 文件夹下

                  # 修改 Nginx 内容

                  # https 配置
	server {
        listen       443 ssl;
        server_name  localhost;

        ssl_certificate      vuedemo.crt; # cert.pem
        ssl_certificate_key  vuedemo.key; # cert.key

        ssl_session_cache    shared:SSL:1m;
        ssl_session_timeout  5m;

        ssl_ciphers  HIGH:!aNULL:!MD5;
        ssl_prefer_server_ciphers  on;

        location / {
            root   html;
            index  index.html index.htm;
        }
    }
# http 配置转发
    server {
        if ($host = example.com) {
            return 301 https://$host$request_uri;
        }


        listen 80 default_server;
        listen [::]:80 default_server;

        server_name _;
        return 404;
    }

                  Windows 终端输入命令重新加载

                  nginx -s reload
